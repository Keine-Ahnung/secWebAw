tralala_users:
 - uid:int (PK)
 - pid:int (FK -> profile)
 - email:text
 - password:text
 - role_id:int (FK)
 - verified:boolean

tralala_profile:
 - pid:int (PK)
 - prename:text
 - surname:text

tralala_roles:
 - role_id:int (PK)
 - role_name:text
 - del_user:boolean
 - set_role:boolean

tralala_posts:
 - post_id:int (PK)
 - uid:int (FK -> user) (oder besser auf profile?)
 - timestamp:date
 - post_text:text
 - hashtags:text
 - upvotes:int
 - downvotes:int

Constraints:

- tralala_users -> tralala_profile:
 ON DELETE CASCADE: Wenn ein User gelöscht wird, soll sein zugehöriges Profil auch gelöscht werden

- tralala_posts -> tralala_users:
 ON DELETE NO ACTION: Wenn ein Post gelöscht wird, soll der referenzierte User nicht gelöscht werden
