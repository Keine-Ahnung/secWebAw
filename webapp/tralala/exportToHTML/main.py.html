102      Ziel des Loginformulars. Überprüft in der Datenbank auf Existenz des angegebenen Benutzers als auch der Übereinstimmung

116          session["logged_in"] 
117          return render_template("Du bist bereits eingeloggt!")

119          pass

122      if request.method == "GET":
